Content of /etc/grub.d/11_windows, 

#!/bin/sh -e
echo "#        Adding Windows GRUB 2 menu"
cat << EOF
menuentry "Windows XP" {
savedefault
insmod chain
insmod ntfs
set root=(hd0,1)
chainloader +1
}
EOF
